    project(test1 C)                                   //ten project ( ten nao cung oke het)

    add_executable(f1 f1.c)                            //lan luot chi ra ten file dich cua tung file ( o day co nghia f1.c bien dich ra ket qua la f1)
    add_executable(f2 f2.c)                            //f2.c bien dich ra ket qua la f2 (co the dat ten khac)
    foreach (p f1 f2)                                  //de chay duoc hai file nay chung ta can them thu vien fdr va igraph ,debugPrintf.
      target_link_libraries(${p} fdr debug_printf)     //them hai thu vien vao bien p ( p la bien dai dien cho f1 va f2) .Ta  them truc tiep hai thu vien fdr va debugPrintf vi no luon luon  duoc bat con thu vien igraph can kiem tra khoa co duoc bat hay khong.
    endforeach (p)                                     //ket thuc viec lien ket thu vien chi hai f1 va f2

    if (USE_IGRAPH)                                    //kiem tra tu thu vien igraph da duoc bat chua (USE_IGRAPH chinh la khoa ta dat trong option trong file cmakelistst.txt o thu muc chinh)
      add_executable(f3 f3.c)                          //tuong tu nhu f1 va f2
      add_executable(f4 f4.c)
      foreach (p f3 f4)                                //dung bien p lam bien dai dien cho hai file f3 va f4
        target_link_libraries(${p} ${IGRAPH_LIBS})     //them thu vien igraph ( IGRAPH_LIBS laf bien duoc khai bao o dong 35 trong file cmakelists.txt trong thu muc chinh)

    endif (USE_IGRAPH)                                 //ket thuc viec kiem tra thu vien
